cmake_minimum_required(VERSION 3.16)
project(Compilerlab3)
set(CMAKE_CXX_STANDARD 14)

# 使用 CMake 的 Flex 和 Bison 模块
find_package(FLEX REQUIRED)
find_package(BISON REQUIRED)

if(FLEX_FOUND)
    message("Info: flex found!")
else()
    message("Error: flex not found!")
endif()

if(BISON_FOUND)
    message("Info: bison found!")
else()
    message("Error: bison not found!")
endif()

# 设置源代码目录 / 生成文件目录
include_directories(${CMAKE_CURRENT_SOURCE_DIR})
include_directories(${CMAKE_CURRENT_BINARY_DIR})

# 指定 Bison 和 Flex 文件的处理
BISON_TARGET(MyParser
             parser.y 
             ${CMAKE_CURRENT_BINARY_DIR}/parser.tab.c
             DEFINES_FILE ${CMAKE_CURRENT_BINARY_DIR}/parser.tab.h
             COMPILE_FLAGS "-Wcounterexamples")
FLEX_TARGET (MyScanner 
             lexer.l 
             ${CMAKE_CURRENT_BINARY_DIR}/lexer.yy.c)
ADD_FLEX_BISON_DEPENDENCY(MyScanner MyParser)

# 指定包含生成的 Bison 和 Flex 输出文件的可执行文件
add_executable(${PROJECT_NAME}
    AST.c
    ${BISON_MyParser_OUTPUTS}
    ${FLEX_MyScanner_OUTPUTS}
)

# 添加编译器警告和语言特性
add_compile_options(-Wall -Wextra -pedantic)

# 链接需要的库
target_compile_features(${PROJECT_NAME} PRIVATE cxx_std_14)
